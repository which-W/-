循环的相关使用
for num in 1 2 3 4 5
do
    echo "the num is $num"
done

for char in "hello world"
do
    echo "the char is $char"
done

将1到100打印出来
for i in `seq 1 100` 或者 $(seq 1 100)
do 
    echo $1
done

for ((i=1 ; i<100; i++))
do
    echo $i
done

i = 1
while(($i < 100))
do 
    echo $i
    (($i++))
done 

i = 1
while(($i < 100))
do
    echo $i 
    let "i++"
done
