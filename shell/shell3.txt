echo的相关参数

echo 执行文件的名称：$0

echo 第一个参数: $1

echo 传递的参数作为一个字符串显示 : $*

echo 传递的参数独立作为每个字符串显示: $@

echo 传递到脚本的参数个数是:$#

echo 最后命令的退出状态: $?

echo 脚本运行的当前进程ID是:$$

sh 1.sh name 可以传递name参数

expr的相关使用

expr 5 + 10 

expr 5 - 10

expr 5 \* 10 乘法

expr 20 / 10 

expr 20 % 10 取余

expr \( 5 + 10 \) \* 7 括号与乘号都需要转译

name = ` expr 5 + 10 ` 使用反引号
echo $name 可以输出结果
