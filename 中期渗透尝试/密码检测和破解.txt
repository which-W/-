针对openssl的破解
echo -n "字符串" | md5sum | awk -F' ' '{print $1}' 转换成md5
echo -n "字符串" | md5sum | awk -F' ' '{print $1}' | od -A n -t x1 转换成md5并转换进制(十六进制)
转换到一行
echo -n "字符串" | md5sum | awk -F' ' '{print $1}' |tr -d '\n' |od -A n -t x1 | tr -d '\n'生成(1)
echo -n '字符串' | openssl enc -d（解密）-a(64base解码) -CipherType（加密类型） -K (1)openssl破解
openssl --help 获取所有加密类型存储到CipherType
sudo awk '{gsub(/ /,"\n");print}' CipherType | sort | uniq > CipherTypes 
循环寻找有关的正确的加密CipherType 
for Cipher in $(cat CipherTypes);do echo -n '字符串' | openssl enc -d -a -$Cipher -K(1) 2>/dev/null;echo $Cipher ;done
试完没有再试一遍没有-n
